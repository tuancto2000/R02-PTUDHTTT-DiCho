CREATE OR ALTER PROCEDURE GetNearestPostCode @east int, @north int
AS
DECLARE @ResultValue VARCHAR(10)
BEGIN 
  SELECT MA_DIA_CHI ,TEN_DIA_CHI , SQRT(
    POWER(CONVERT(bigint,(TOA_DO_DONG - @east)), 2) +
    POWER(CONVERT(bigint,(TOA_DO_BAC - @north)), 2)) AS distance
FROM DIA_CHI WHERE SQRT(
    POWER(CONVERT(bigint,(TOA_DO_DONG - @east)), 2) +
    POWER(CONVERT(bigint,(TOA_DO_BAC - @north)), 2))  < 100000 ORDER BY distance

END
GO
CREATE OR ALTER trigger UpdateDonGia_ChiTietDonHang
ON dbo.CHI_TIET_DON_HANG
AFTER INSERT, UPDATE
AS 
BEGIN

    --
    -- Check if this is an INSERT, UPDATE or DELETE Action.
    -- 
    DECLARE @gia as int;
	DECLARE @machitietdonhang as int;
	DECLARE @madonhang as int;
	
	SET @madonhang = (select MA_DON_HANG from inserted)
	SET @machitietdonhang = (select id from inserted)
	SET @gia = (select SAN_PHAM.GIA_SP FROM inserted join SAN_PHAM on inserted.MA_SP = SAN_PHAM.MA_SP where SAN_PHAM.MA_SP = inserted.MA_SP)

	SET TongTien = (SELECT SUM(DON_GIA) FROM CHI_TIET_DON_HANG where MA_DON_HANG = @madonhang)
	where MA_DON_HANG = @madonhang


END;
GO

CREATE OR ALTER trigger Update_DanhGia
ON dbo.DANH_GIA
AFTER INSERT, UPDATE
AS 
BEGIN

    --
    -- Check if this is an INSERT, UPDATE or DELETE Action.
    -- 
    
	DECLARE @masp as int;
	
	SET @masp = (select MA_SAN_PHAM from inserted)
	
	UPDATE SAN_PHAM
	SET SoLuotDanhGia = SoLuotDanhGia + 1
	where MA_SP = @masp;
	UPDATE SAN_PHAM
	SET TrungBinhSao = (select SUM(SAO) from DANH_GIA where MA_SAN_PHAM = @masp)*1.00 / SoLuotDanhGia 
	where MA_SP = @masp



END;
GO


CREATE OR ALTER trigger Delete_DanhGia
ON dbo.DANH_GIA

AS 
BEGIN

    --
    -- Check if this is an INSERT, UPDATE or DELETE Action.
    --

	SET @masp = (select MA_SAN_PHAM from deleted)
	
	UPDATE SAN_PHAM

	SET TrungBinhSao = (select SUM(SAO) from DANH_GIA where MA_SAN_PHAM = @masp)*1.00  / SoLuotDanhGia 
	where MA_SP = @masp



END;
GO
